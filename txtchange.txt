def load_medical_txt(file_path):
    documents = []
    
    # Open and read the plain text file
    with open(file_path, 'r', encoding='utf-8') as file:
        text_content = file.read()
        
    # Pack the entire text into a standard LangChain Document object
    # We add metadata so the system can still provide a source citation
    doc = Document(
        page_content=text_content,
        metadata={
            "source": os.path.basename(file_path),
            "type": "Text Documentation"
        }
    )
    documents.append(doc)
    return documents
